HTML-specific filters for use in .html page formats.

htmltable

Generate a HTML <table/> from a Single or List of Dictionaries. headerStyle can be any Text Style.

{{ 'live-template' | partial({ template: `{{ [{FirstName: 'Kurt', Age: 27},{FirstName: 'Jimi', Age: 27}] | assignTo: rockstars }} {{ rockstars | htmltable }} Uses defaults: {{ rockstars | htmltable({ headerTag: 'th', headerStyle: 'splitCase' }) }}` }) }}
htmltable customizations
{{ 'live-template' | partial({ rows:3, template: `{{ [{FirstName: 'Kurt', Age: 27},{FirstName: 'Jimi', Age: 27}] | assignTo: rockstars }} {{ rockstars | htmltable({ className: "table table-striped", caption: "Rockstars" }) }} {{ [] | htmltable({ emptyCaption: "No Rocksars"}) }}` }) }} {{ "doc-links" | partial({ order }) }}
